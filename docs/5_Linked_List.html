New Exercise!!

A loop in a linked list occurs when a node’s next pointer points to a previous node, forming a cycle. Write a method detect_loop() that checks if a linked list contains a loop. If a loop is detected, return True. Otherwise, return False.

class Node:
    def __init__(self, value):
        self.value = value
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def add_node(self, value):
        new_node = Node(value)
        if not self.head:
            self.head = new_node
            return

        current = self.head
        while current.next:
            current = current.next
        current.next = new_node

    def create_loop(self, index):
        """
        Creates a loop by connecting the last node to the node at the given index.
        """
        if not self.head:
            return

        loop_node = None
        current = self.head
        count = 0

        while current.next:
            if count == index:
                loop_node = current
            current = current.next
            count += 1

        if loop_node:
            current.next = loop_node

    def detect_loop(self):
        slow = self.head
        fast = self.head

        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next

            if slow == fast:
                return True

        return False

# Example Usage
ll = LinkedList()
ll.add_node(1)
ll.add_node(2)
ll.add_node(3)
ll.add_node(4)
ll.add_node(5)

ll.create_loop(2)  # Creates a loop: 5 -> 3

print("Loop Detected:", ll.detect_loop())  # Expected Output: True

ll2 = LinkedList()
ll2.add_node(1)
ll2.add_node(2)
ll2.add_node(3)

print("Loop Detected:", ll2.detect_loop())  # Expected Output: False
